# dotfiles-Kali :: base CLI stack (apt / Debian names)
# One package per line. '#' comments and blank lines are ignored by bootstrap.sh.
# Note: on Debian/Kali, `bat` ships the binary `batcat` and `fd-find` ships
# `fdfind` — Core's tools.zsh already resolves those alt binary names.

# shell + multiplexer
zsh
tmux

# modern coreutils replacements
eza
bat            # binary: batcat
fd-find        # binary: fdfind
ripgrep
zoxide
fzf
git-delta      # binary: delta
btop
tealdeer       # tldr client (binary: tldr); swap to `tldr` if unavailable

# the rest of core-doctor's modern-CLI set (apt/Debian names)
du-dust        # binary: dust — the package is `du-dust`, the command stays `dust`
duf            # disk-usage/free viewer
procs          # modern `ps`
sd             # find & replace (src: rust-sd); binary: sd
gron           # greppable JSON
xh             # HTTPie-style HTTP client (src: rust-xh); binary: xh
yq-go          # mikefarah's Go `yq` — Debian ALSO ships `yq` (Python kislyuk);
               # we want the Go one, so use `yq-go` and NOT plain `yq`.
glow           # markdown renderer — recent (Debian sid → Kali rolling); may lag migration
gum            # charmbracelet TUI toolkit — recent (Debian sid → Kali rolling); may lag migration
# core-doctor also wants: doggo, carapace, sesh, op — those are go/vendor and are
# installed by bootstrap.sh (not in apt / not reliably packaged), not listed here.

# editor + vcs
neovim
git
openssh-client
git-credential-manager   # optional; remove if not packaged on your box

# clipboard (mostly inert on WSL — Core uses clip.exe — but harmless)
wl-clipboard
xclip

# build + language providers
build-essential
cargo
python3-pip
python3-venv
python3-pynvim          # nvim python provider
tree-sitter-cli         # generate/build/test tree-sitter parsers (nvim-treesitter)

# everyday utilities
curl
wget
jq
gawk
direnv
unzip
